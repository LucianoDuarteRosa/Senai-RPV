create database tarefas;
use tarefas;

CREATE TABLE Tarefas (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    Nome VARCHAR(255) NOT NULL,
    Descricao VARCHAR(255) NOT NULL,
    Data_Inicio DATE NOT NULL,
    Data_Termino DATE,
    Finalizado BOOLEAN default false
);
